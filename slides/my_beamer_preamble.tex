

% define colors 
\usepackage{color}
  \definecolor{myviolet}{HTML}{3333B2}
  \definecolor{palegray}{HTML}{D7DBDD}
  
  
  
% reset the beamer template
\usetheme{default} 

% omit navigation symbols
\beamertemplatenavigationsymbolsempty 
		
% set color for title slide main title and headline
\setbeamercolor{title}{fg = myviolet}
\setbeamercolor{frametitle}{fg = myviolet}

% typeface settings
\usepackage[latin1]{inputenc}
\usepackage{times} 
\usepackage[T1]{fontenc} 

% format lists
\usepackage{enumitem}
\setlist[itemize]{label = \color{myviolet}$\blacktriangleright$, itemsep = 1\baselineskip, leftmargin = *, font=\large}

% set parameters for the PDF file
\hypersetup{
pdftitle = {\talktitle},
pdfauthor = {\talkauthor},
pdfkeywords = {reproducible research},
pdfnewwindow = true,  % links in new PDF window
colorlinks = true,    % false: boxed links; true: colored links
linkcolor = myviolet, % internal links (change box color w/ linkbordercolor)
citecolor = myviolet, % color of links to bibliography
filecolor = myviolet, % color of file links
urlcolor = myviolet   % color of external links
}

% custom footline with note on left and page number on the right
\makeatother%
\setbeamertemplate{footline}%
{%
\leavevmode%
% the {12} linespacing option does not seem to work
\hbox{\fontsize{7}{18}\selectfont% 
% on the left, I have a space for permissions note or full citation
\begin{beamercolorbox}[wd = 0.88\paperwidth, ht = 5ex, dp = 3mm, leftskip = 3mm, rightskip = 0mm]{}\color{gray}\foottext%
\end{beamercolorbox}% 
% on the right is the slide number
\begin{beamercolorbox}[wd = 0.12\paperwidth, ht = 5ex, dp = 3mm, leftskip = 0mm, rightskip = 3mm]{}\hfill\insertpagenumber%
\end{beamercolorbox}% 
}%
\vskip0pt%
}%
\makeatletter%
